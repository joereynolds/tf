# Testing a word gets converted to uppercase
$ echo "test" | ./dist/tf upper
TEST

# Testing a word gets converted to lowercase
$ echo "GO LOWER PLEASE" | ./dist/tf lower
go lower please

# Testing a word gets converted to titlecase
$ echo "hello james bond" | ./dist/tf title
Hello James Bond

# Testing a string join
$ echo "I am joining strings" | ./dist/tf join -
I-am-joining-strings

# Testing an nth word replacement
$ echo "I am joining strings" | ./dist/tf nth word 2 foo
I foo joining foo
